79 void
80 e_narrowResonanceCrossSection::crossSectionCalculation(const double) // _bwnormsave (unused)
81 {
82  // This subroutine calculates the vector meson cross section assuming
83  // a narrow resonance. For reference, see STAR Note 386.
84 
85  double W, dEgamma, minEgamma;
86  double ega[3] = {0};
87  double int_r,dR;
88  double int_r2, dR2;
89  int iEgamma, nEgamma,beam;
90 
91  //Integration is done with exponential steps, in target frame
92  //nEgamma = _VMnumEgamma;
93  nEgamma = 1000;
94  dEgamma = std::log(_targetMaxPhotonEnergy/_targetMinPhotonEnergy)/nEgamma;
95  minEgamma = std::log(_targetMinPhotonEnergy);
96 
97  cout<<" Using Narrow Resonance ..."<<endl;
98 
99  W = getChannelMass();
100  //
101  printf(" gamma+nucleon threshold (CMS): %e GeV \n", _cmsMinPhotonEnergy);
102 
103  int A_1 = getbbs().beam1().A();
104  int A_2 = getbbs().beam2().A();
105 
106  if( A_2 == 0 && A_1 >= 1 ){
107  // pA, first beam is the nucleus and is in this case the target
108  beam = 1;
109  } else if( A_1 ==0 && A_2 >= 1){
110  // photon energy in Target frame
111  beam = 2;
112  } else {
113  // photon energy in Target frame
114  beam = 2;
115  }
116 
117  int_r=0.;
118  int_r2= 0;
119  // Do this first for the case when the first beam is the photon emitter
120  // The variable beam (=1,2) defines which nucleus is the target
121  // Target beam ==2 so repidity is negative. Can generalize later
122  // These might be useful in a future iteration
123  int nQ2 = 1000;
124  printf(" gamma+nucleon threshold (Target): %e GeV \n", _targetMinPhotonEnergy);
125  for(iEgamma = 0 ; iEgamma < nEgamma; ++iEgamma){ // Integral over photon energy
126  // Target frame energies
127  ega[0] = exp(minEgamma + iEgamma*dEgamma );
128  ega[1] = exp(minEgamma + (iEgamma+1)*dEgamma );
129  ega[2] = 0.5*(ega[0]+ega[1]);
130 
131  // Integral over Q2
132  double dndE[3] = {0}; // Effective photon flux
133  double full_int[3] = {0}; // Full e+X --> e+X+V.M. cross section
134  //
135  for( int iEgaInt = 0 ; iEgaInt < 3; ++iEgaInt){ // Loop over the energies for the three points to integrate over Q2
136  //These are the physical limits
137  double Q2_min = std::pow(starlightConstants::mel*ega[iEgaInt],2.0)/(_electronEnergy*(_electronEnergy-ega[iEgaInt]));
138  double Q2_max = 4.*_electronEnergy*(_electronEnergy-ega[iEgaInt]);
139  if(_useFixedRange == true){
140  if( Q2_min < _gammaMinQ2 )
141  Q2_min = _gammaMinQ2;
142  if( Q2_max > _gammaMaxQ2 )
143  Q2_max = _gammaMaxQ2;
144  }
145  double lnQ2ratio = std::log(Q2_max/Q2_min)/nQ2;
146  double lnQ2_min = std::log(Q2_min);
147  //
148  int q2end = -99;
149  for( int iQ2 = 0 ; iQ2 < nQ2; ++iQ2){ // Integral over photon virtuality
150  //
151  double q2_1 = exp( lnQ2_min + iQ2*lnQ2ratio);
152  double q2_2 = exp( lnQ2_min + (iQ2+1)*lnQ2ratio);
153  double q2_12 = (q2_2+q2_1)/2.;
154  // Integrating the effective photon flux
155  dndE[iEgaInt] +=(q2_2-q2_1)*( getcsgA_Q2_dep(q2_1)*photonFlux(ega[iEgaInt],q2_1)
156  +getcsgA_Q2_dep(q2_2)*photonFlux(ega[iEgaInt],q2_2)
157  +4.*getcsgA_Q2_dep(q2_12)*photonFlux(ega[iEgaInt],q2_12) );
158  // Integrating cross section
159  full_int[iEgaInt] += (q2_2-q2_1)*( g(ega[iEgaInt],q2_1)*getcsgA(ega[iEgaInt],q2_1,beam)
160  + g(ega[iEgaInt],q2_2)*getcsgA(ega[iEgaInt],q2_2,beam)
161  + 4.*g(ega[iEgaInt],q2_12)*getcsgA(ega[iEgaInt],q2_12,beam) );
162  }
163  q2end = -99 ;
164  // Finish the Q2 integration for the three end-points (Siumpsons rule)
165  dndE[iEgaInt] = dndE[iEgaInt]/6.;
166  full_int[iEgaInt] = full_int[iEgaInt]/6.;
167  }
168  // Finishing cross-section integral
169  dR = full_int[0];
170  dR += full_int[1];
171  dR += 4.*full_int[2];
172  dR = dR*(ega[1]-ega[0])/6.;
173 
174  // Finishing integral over the effective photon flux
175  dR2 = dndE[0];
176  dR2 += dndE[1];
177  dR2 += 4.*dndE[2];
178  dR2 = dR2*(ega[1]-ega[0])/6.;
179  //
180  int_r = int_r+dR;
181  int_r2 = int_r2 + dR2;
182  }
183  cout<<endl;
184  if(_useFixedRange == true){
185  cout<<" Using fixed Q2 range "<<_gammaMinQ2 << " < Q2 < "<<_gammaMaxQ2<<endl;
186  }
187  printCrossSection(" Total cross section: ",int_r);
188  // printCrossSection(" Total effective photon flux ", int_r2); // These are used to obtain the photon flux and gammaX --> VM X x-section
189  // printCrossSection(" gamma+X --> VM+X ", int_r/int_r2); // commented out for the mean time, not necesary in current implementation
190  //
191  //cout<<endl;
192  setPhotonNucleusSigma(0.01*int_r);
193  #ifdef _makeGammaPQ2_
194  makeGammaPQ2dependence();
195  #endif
196  cout<<"Here"<<endl;
197 }
