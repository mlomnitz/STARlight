73 void
74 e_wideResonanceCrossSection::crossSectionCalculation(const double bwnormsave)
75 {
76  // This subroutine calculates the cross-section assuming a wide
77  // (Breit-Wigner) resonance.
78 
79  double W,dW, dEgamma, minEgamma;
80  double ega[3] = {0};
81  double int_r,dR;
82  double Eth;
83  int iW,nW,iEgamma,nEgamma,beam;
84 
85  double bwnorm = bwnormsave; //used to transfer the bwnorm from the luminosity tables
86 
87  //gamma+nucleon threshold.
88  Eth=0.5*(((_wideWmin+protonMass)*(_wideWmin+protonMass)
89  -protonMass*protonMass)/(_Ep+sqrt(_Ep*_Ep-protonMass*protonMass)));
90 
91  // For W integration
92  nW = 100;
93  dW = (_wideWmax-_wideWmin)/double(nW);
94  // For Egamma integration
95  nEgamma = 1000;
96  dEgamma = std::log(_targetMaxPhotonEnergy/_targetMinPhotonEnergy)/nEgamma;
97  minEgamma = std::log(_targetMinPhotonEnergy);
98 
99 
100  if (getBNORM() == 0.){
101  cout<<" Using Breit-Wigner Resonance Profile."<<endl;
102  }
103  else{
104  cout<<" Using Breit-Wigner plus direct pi+pi- profile."<<endl;
105  }
106 
107  cout<<" Integrating over W from "<<_wideWmin<<" to "<<_wideWmax<<endl;
108 
109  int A_1 = getbbs().beam1().A();
110  int A_2 = getbbs().beam2().A();
111 
112  if( A_2 == 0 && A_1 >= 1 ){
113  // eA, first beam is the nucleus and is in this case the target
114  beam = 1;
115  } else if( A_1 ==0 && A_2 >= 1){
116  // eA, second beam is the nucleus and is in this case the target
117  beam = 2;
118  } else {
119  beam = 2;
120  }
121 
122  int_r=0.;
123 
124  // Do this first for the case when the first beam is the photon emitter
125  // The variable beam (=1,2) defines which nucleus is the target
126  // Integration done using Simpson's rule
127  for(iW=0;iW<=nW-1;iW++){
128 
129  W = _wideWmin + double(iW)*dW + 0.5*dW;
130  int nQ2 = 1000;
131  for(iEgamma = 0 ; iEgamma < nEgamma; ++iEgamma){ // Integral over photon energy
132  // Target frame photon energies
133  ega[0] = exp(minEgamma + iEgamma*dEgamma );
134  ega[1] = exp(minEgamma + (iEgamma+1)*dEgamma );
135  ega[2] = 0.5*(ega[0]+ega[1]);
136  // Integral over Q2
137  double full_int[3] = {0}; // Full e+X --> e+X+V.M. cross section
138  //
139  for( int iEgaInt = 0 ; iEgaInt < 3; ++iEgaInt){ // Loop over the energies for the three points to integrate over Q2
140  //These are the physical limits
141  double Q2_min = std::pow(starlightConstants::mel*ega[iEgaInt],2.0)/(_electronEnergy*(_electronEnergy-ega[iEgaInt]));
142  double Q2_max = 4.*_electronEnergy*(_electronEnergy-ega[iEgaInt]);
143  if(_useFixedRange == true){
144  if( Q2_min < _gammaMinQ2 )
145  Q2_min = _gammaMinQ2;
146  if( Q2_max > _gammaMaxQ2 )
147  Q2_max = _gammaMaxQ2;
148  }
149  double lnQ2ratio = std::log(Q2_max/Q2_min)/nQ2;
150  double lnQ2_min = std::log(Q2_min);
151  //
152  int q2end = -99;
153  for( int iQ2 = 0 ; iQ2 < nQ2; ++iQ2){ // Integral over photon virtuality
154  //
155  double q2_1 = exp( lnQ2_min + iQ2*lnQ2ratio);
156  double q2_2 = exp( lnQ2_min + (iQ2+1)*lnQ2ratio);
157  double q2_12 = (q2_2+q2_1)/2.;
158  //
159  // Integrating cross section
160  full_int[iEgaInt] += (q2_2-q2_1)*( g(ega[iEgaInt],q2_1)*getcsgA(ega[iEgaInt],q2_1,beam)
161  + g(ega[iEgaInt],q2_2)*getcsgA(ega[iEgaInt],q2_2,beam)
162  + 4.*g(ega[iEgaInt],q2_12)*getcsgA(ega[iEgaInt],q2_12,beam) );
163  }
164  q2end = -99 ;
165  full_int[iEgaInt] = full_int[iEgaInt]/6.;
166  }
167  // Finishing cross-section integral
168  dR = full_int[0];
169  dR += full_int[1];
170  dR += 4.*full_int[2];
171  dR = dR*(ega[1]-ega[0])/6.;
172  int_r = int_r + dR*breitWigner(W,bwnorm)*dW;
173  }
174  }
175  cout<<endl;
176  if (0.01*int_r > 1.){
177  cout<< " Total cross section: "<<0.01*int_r<<" barn."<<endl;
178  } else if (10.*int_r > 1.){
179  cout<< " Total cross section: " <<10.*int_r<<" mb."<<endl;
180  } else if (10000.*int_r > 1.){
181  cout<< " Total cross section: " <<10000.*int_r<<" microb."<<endl;
182  } else if (10000000.*int_r > 1.){
183  cout<< " Total cross section: " <<10000000.*int_r<<" nanob."<<endl;
184  } else if (1.E10*int_r > 1.){
185  cout<< " Total cross section: "<<1.E10*int_r<<" picob."<<endl;
186  } else {
187  cout<< " Total cross section: " <<1.E13*int_r<<" femtob."<<endl;
188  }
189  cout<<endl;
190  setPhotonNucleusSigma(0.01*int_r);
191  //
192 #ifdef _makeGammaPQ2_
193  makeGammaPQ2dependence(bwnormsave);
194 #endif
195 }

223  cout<<" Using Breit-Wigner Resonance Profile."<<endl;
224  }

226  cout<<" Using Breit-Wigner plus direct pi+pi- profile."<<endl;
227  }
228

231  //Lomnitz old used for XX
232  Eth=0.5*(((W+protonMass)*(W+protonMass)-
233  protonMass*protonMass)/(_Ep+sqrt(_Ep*_Ep-protonMass*protonMass)));
234  // Adapted for eX
235  //Eth=0.5*(((W+starlightConstants::mel)*(W +starlightConstants::mel)-
236  // starlightConstants::mel*starlightConstants::mel)/(_electronEnergy+sqrt(_electronEnergy*_electronEnergy-starlightConstants::mel*starlightConstants::mel)));

238  printf(" gamma+nucleon threshold: %e GeV \n", Eth);

241  int A_2 = getbbs().beam2().A();
242 
243 
244  // Do this first for the case when the first beam is the photon emitter
245  // The variable beam (=1,2) defines which nucleus is the target
246  // Target beam ==2 so repidity is negative. Can generalize later
248  cout<<" Lomnitz debug :: sigma_gamma_p --> VM_p "<<endl;
249  cout<<" Q2+MV2 \t \t"<<" sigma_gamma_p --> VM_p (nanob)"<<endl;
250  double target_cm = acosh(_target_beamLorentz);
251  // another - sign from subraction in addition rule
252  double exp_target_cm = exp(-target_cm);
253  double int_r2;
254  for( int iQ2 = 0 ; iQ2 < nQ2bins; ++iQ2){
255  int_r=0.;
256  int_r2=0.;
257  //
258  double q2_cor = getcsgA_Q2_dep( (q2Edge[iQ2+1] + q2Edge[iQ2])/2. );
259  for(I=0;I<=NW-1;I++){
260 
261  W = _wideWmin + double(I)*dW + 0.5*dW;
262  for(J=0;J<=(NY-1);J++){
263  y1 = _wideYmin + double(J)*dY;
264  y2 = _wideYmin + double(J+1)*dY;
265  y12 = 0.5*(y1+y2);
266  double target_ega1, target_ega2, target_ega12;
267  if( A_2 == 0 && A_1 >= 1 ){
268  // pA, first beam is the nucleus and is in this case the target
269  ega1 = 0.5*W*exp(-y1);
270  ega2 = 0.5*W*exp(-y2);
271  ega12 = 0.5*W*exp(-y12);
272  beam = 1;
273  } else if( A_1 ==0 && A_2 >= 1){
274  // pA, second beam is the nucleus and is in this case the target
275  ega1 = 0.5*W*exp(y1);
276  ega2 = 0.5*W*exp(y2);
277  ega12 = 0.5*W*exp(y12);
278  // photon energy in Target frame
279  beam = 2;
280  } else {
281  ega1 = 0.5*W*exp(y1);
282  ega2 = 0.5*W*exp(y2);
283  ega12 = 0.5*W*exp(y12);
284  // photon energy in Target frame
285  beam = 2;
286  }
287  //
288  if(ega1 < Eth || ega2 < Eth)
289  continue;
290  if(ega2 > maxPhotonEnergy() || ega1 > maxPhotonEnergy() )
291  continue;
292  target_ega1 = ega1*exp_target_cm;
293  target_ega12 = ega12*exp_target_cm;
294  target_ega2 = ega2*exp_target_cm;
295  //cout<<"Nortmalizations "<<integrated_x_section(ega1,0,50)<<endl;
296  //
297  csgA1=getcsgA(ega1,W,beam);
298  double full_range_1 = integrated_x_section(target_ega1);
299  // >> Middle Point =====>>>
300  csgA12=getcsgA(ega12,W,beam);
301  double full_range_12 = integrated_x_section(target_ega12);
302  // >> Second Point =====>>>
303  csgA2=getcsgA(ega2,W,beam);
304  double full_range_2 = integrated_x_section(target_ega2);
305  //
306 
307  //>> Sum the contribution for this W,Y. The 2 accounts for the 2 beams
308  dR = q2_cor*csgA1;
309  dR = dR + 4.*q2_cor*csgA12;
310  dR = dR + q2_cor*csgA2;
311  dR = dR*(dY/6.)*breitWigner(W,bwnorm)*dW;
312  //
313  dR2 = full_range_1*csgA1;
314  dR2 = dR2 + 4.*full_range_12*csgA12;
315  dR2 = dR2 + full_range_2*csgA2;
316  dR2 = dR2*(dY/6.)*breitWigner(W,bwnorm)*dW;
317  //
318  int_r = int_r+dR;
319  int_r2 = int_r2 +dR2;
320  }
321  //
322  }
323  if( iQ2 ==0 )
324  cout<<"Full range "<<int_r2*10000000<<endl;
325  cout<<(q2Edge[iQ2+1]+q2Edge[iQ2])/2.+W*W<<" , "<<10000000.*int_r<<endl;
326  }
