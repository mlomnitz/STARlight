83 bool
84 starlight::init()
85 {
86  if(Starlight_VERSION_MAJOR == 9999)
87  {
88  cout << endl << "#########################################" << endl
89  << " Initialising Starlight version: trunk..." << endl
90  << "#########################################" << endl << endl;
91  }
92  else
93  {
94  cout << endl << "#########################################" << endl
95  << " Initialising Starlight version: " << Starlight_VERSION_MAJOR << "."
96  << Starlight_VERSION_MINOR << "." << Starlight_VERSION_MINOR_MINOR << "..." << endl
97  << "#########################################" << endl << endl;
98  }
99 
100  _nmbEventsPerFile = _inputParameters->nmbEvents(); // for now we write only one file...
101 
102  _beamSystem = new beamBeamSystem(*_inputParameters);
103 
104  // This sets the precision used by cout for printing
105  cout.setf(ios_base::fixed,ios_base::floatfield);
106  cout.precision(3);
107 
108  std::string _baseFileName;
109  _baseFileName = _inputParameters->baseFileName();
110  _lumLookUpTableFileName = _baseFileName + ".txt";
111 
112  const bool lumTableIsValid = luminosityTableIsValid();
113 
114  // Do some sanity checks of the input parameters here.
115  if( _inputParameters->beam1Z() > _inputParameters->beam1A() ){
116  printErr << endl << "A must be >= Z; A beam1 = "<<_inputParameters->beam1A()<<", Z beam1 = "<<_inputParameters->beam1Z()<<". Terminating."<<endl ;
117  return false;
118  }
119  if( _inputParameters->beam2Z() > _inputParameters->beam2A() ){
120  printErr << endl << "A must be >= Z; A beam2 = "<<_inputParameters->beam2A()<<", Z beam2 = "<<_inputParameters->beam2Z()<<". Terminating."<<endl ;
121  return false;
122  }
123  if( _inputParameters->interactionType() == PHOTONPOMERONINCOHERENT && _inputParameters->beam1A() == 1 &&
124  _inputParameters->beam1Z() == 1 && _inputParameters->beam2A() == 1 && _inputParameters->beam2Z() ){
125  printErr << endl << " Do not use PROD_MODE = 4 for pp collisions. Use PROD_MODE = 2 or 3 instead. Terminating."<<endl;
126  return false;
127  }
128 
129  bool createChannel = true;
130  switch (_inputParameters->interactionType()) {
131  case PHOTONPHOTON:
132  if (!lumTableIsValid) {
133  printInfo << "creating luminosity table for photon-photon channel" << endl;
134  twoPhotonLuminosity(*_inputParameters, _beamSystem->beam1(), _beamSystem->beam2());
135  }
136  break;
137  case PHOTONPOMERONNARROW: // narrow and wide resonances use
138  case PHOTONPOMERONWIDE: // the same luminosity function
139  if (!lumTableIsValid) {
140  printInfo << "creating luminosity table for coherent photon-Pomeron channel" <<endl;
141  photonNucleusLuminosity lum(*_inputParameters, *_beamSystem);
142  }
143  break;
144  case PHOTONPOMERONINCOHERENT: // narrow and wide resonances use
145  if (!lumTableIsValid) {
146  printInfo << "creating luminosity table for incoherent photon-Pomeron channel" << endl;
147  incoherentPhotonNucleusLuminosity lum(*_inputParameters, *_beamSystem);
148  }
149  break;
150 #ifdef ENABLE_DPMJET
151  case PHOTONUCLEARSINGLE:
152  createChannel = false;
153  _eventChannel = new starlightDpmJet(*_inputParameters, *_beamSystem);
154  std::cout << "CREATING PHOTONUCLEAR/DPMJET SINGLE" << std::endl;
155  dynamic_cast<starlightDpmJet*>(_eventChannel)->setSingleMode();
156  dynamic_cast<starlightDpmJet*>(_eventChannel)->setMinGammaEnergy(_inputParameters->minGammaEnergy());
157  dynamic_cast<starlightDpmJet*>(_eventChannel)->setMaxGammaEnergy(_inputParameters->maxGammaEnergy());
158  dynamic_cast<starlightDpmJet*>(_eventChannel)->init();
159  break;
160  case PHOTONUCLEARDOUBLE:
161  createChannel = false;
162  _eventChannel = new starlightDpmJet(*_inputParameters, *_beamSystem);
163  std::cout << "CREATING PHOTONUCLEAR/DPMJET DOUBLE" << std::endl;
164  dynamic_cast<starlightDpmJet*>(_eventChannel)->setDoubleMode();
165  dynamic_cast<starlightDpmJet*>(_eventChannel)->setMinGammaEnergy(_inputParameters->minGammaEnergy());
166  dynamic_cast<starlightDpmJet*>(_eventChannel)->setMaxGammaEnergy(_inputParameters->maxGammaEnergy());
167  dynamic_cast<starlightDpmJet*>(_eventChannel)->init();
168  break;
169  case PHOTONUCLEARSINGLEPA:
170  createChannel = false;
171  _eventChannel = new starlightDpmJet(*_inputParameters, *_beamSystem);
172  std::cout << "CREATING PHOTONUCLEAR/DPMJET SINGLE" << std::endl;
173  dynamic_cast<starlightDpmJet*>(_eventChannel)->setSingleMode();
174  dynamic_cast<starlightDpmJet*>(_eventChannel)->setProtonMode();
175  dynamic_cast<starlightDpmJet*>(_eventChannel)->setMinGammaEnergy(_inputParameters->minGammaEnergy());
176  dynamic_cast<starlightDpmJet*>(_eventChannel)->setMaxGammaEnergy(_inputParameters->maxGammaEnergy());
177  dynamic_cast<starlightDpmJet*>(_eventChannel)->init();
178  break;
179 #endif
180 #ifdef ENABLE_PYTHIA6
181  case PHOTONUCLEARSINGLEPAPY:
182  createChannel = false;
183  _eventChannel = new starlightPythia(*_inputParameters, *_beamSystem);
184  std::cout << "CREATING PHOTONUCLEAR/PYTHIA SINGLE" << std::endl;
185  dynamic_cast<starlightPythia*>(_eventChannel)->setSingleMode();
186  dynamic_cast<starlightPythia*>(_eventChannel)->setMinGammaEnergy(_inputParameters->minGammaEnergy());
187  dynamic_cast<starlightPythia*>(_eventChannel)->setMaxGammaEnergy(_inputParameters->maxGammaEnergy());
188  dynamic_cast<starlightPythia*>(_eventChannel)->init(_inputParameters->pythiaParams(), _inputParameters->pythiaFullEventRecord());
189  break;
190 #endif
191  default:
192  {
193  printWarn << "unknown interaction type '" << _inputParameters->interactionType() << "'."
194  << " cannot initialize starlight." << endl;
195  return false;
196  }
197  }
198 
199  if(createChannel)
200  {
201  if (!createEventChannel())
202  return false;
203  }
204 
205  _isInitialised = true;
206  return true;
207 }

221 bool
222 starlight::luminosityTableIsValid() const
223 {
224  printInfo << "using random seed = " << _inputParameters->randomSeed() << endl;
225 
226  ifstream lumLookUpTableFile(_lumLookUpTableFileName.c_str());
227  lumLookUpTableFile.precision(15);
228  if ((!lumLookUpTableFile) || (!lumLookUpTableFile.good())) {
229  return false;
230  }
231 
232  unsigned int beam1Z, beam1A, beam2Z, beam2A;
233  double beamLorentzGamma = 0;
234  double maxW = 0, minW = 0;
235  unsigned int nmbWBins;
236  double maxRapidity = 0;
237  unsigned int nmbRapidityBins;
238  int productionMode, beamBreakupMode;
239  bool interferenceEnabled = false;
240  double interferenceStrength = 0;
241  bool coherentProduction = false;
242  double incoherentFactor = 0, deuteronSlopePar = 0, maxPtInterference = 0;
243  int nmbPtBinsInterference;
244  std::string validationKey;
245  if (!(lumLookUpTableFile
246  >> validationKey
247  >> beam1Z >> beam1A
248  >> beam2Z >> beam2A
249  >> beamLorentzGamma
250  >> maxW >> minW >> nmbWBins
251  >> maxRapidity >> nmbRapidityBins
252  >> productionMode
253  >> beamBreakupMode
254  >> interferenceEnabled >> interferenceStrength
255  >> maxPtInterference
256  >> nmbPtBinsInterference
257  >> coherentProduction >> incoherentFactor
258  >> deuteronSlopePar))
259  // cannot read parameters from lookup table file
260  return false;
261 
262  std::string validationKeyEnd;
263  while(!lumLookUpTableFile.eof())
264  {
265  lumLookUpTableFile >> validationKeyEnd;
266  }
267  lumLookUpTableFile.close();
268  return (validationKey == _inputParameters->parameterValueKey() && validationKeyEnd == validationKey);
269  return true;
270 }

273 bool
274 starlight::createEventChannel()
275 {
276  switch (_inputParameters->prodParticleType()) {
277  case ELECTRON:
278  case MUON:
279  case TAUON:
280  case TAUONDECAY:
281  {
282  _eventChannel = new Gammagammaleptonpair(*_inputParameters, *_beamSystem);
283  if (_eventChannel)
284  return true;
285  else {
286  printWarn << "cannot construct Gammagammaleptonpair event channel." << endl;
287  return false;
288  }
289  }
290  case A2: // jetset/pythia
291  case ETA: // jetset/pythia
292  case ETAPRIME: // jetset/pythia
293  case ETAC: // jetset/pythia
294  case F0: // jetset/pythia
295  {
296 #ifdef ENABLE_PYTHIA
297  // PythiaOutput = true;
298  cout<<"Pythia is enabled!"<<endl;
299 // return true;
300 #else
301  printWarn << "Starlight is not compiled against Pythia8; "
302  << "jetset event channel cannot be used." << endl;
303  return false;
304 #endif
305  }
306  case F2:
307  case F2PRIME:
308  case ZOVERZ03:
309  case AXION: // AXION HACK
310  {
311  // #ifdef ENABLE_PYTHIA
312  cout<<" This is f2, f2prim, rho^0 rho^0, or axion "<<endl;
313  _eventChannel= new Gammagammasingle(*_inputParameters, *_beamSystem);
314  if (_eventChannel)
315  return true;
316  else {
317  printWarn << "cannot construct Gammagammasingle event channel." << endl;
318  return false;
319  }
320  }
321  case RHO:
322  case RHOZEUS:
323  case FOURPRONG:
324  case OMEGA:
325  case PHI:
326  case JPSI:
327  case JPSI2S:
328  case JPSI2S_ee:
329  case JPSI2S_mumu:
330  case JPSI_ee:
331  case JPSI_mumu:
332  case UPSILON:
333  case UPSILON_ee:
334  case UPSILON_mumu:
335  case UPSILON2S:
336  case UPSILON2S_ee:
337  case UPSILON2S_mumu:
338  case UPSILON3S:
339  case UPSILON3S_ee:
340  case UPSILON3S_mumu:
341  {
342  if (_inputParameters->interactionType() == PHOTONPOMERONNARROW) {
343  _eventChannel = new Gammaanarrowvm(*_inputParameters, *_beamSystem);
344  if (_eventChannel)
345  return true;
346  else {
347  printWarn << "cannot construct Gammaanarrowvm event channel." << endl;
348  return false;
349  }
350  }
351 
352  if (_inputParameters->interactionType() == PHOTONPOMERONWIDE) {
353  _eventChannel = new Gammaawidevm(*_inputParameters, *_beamSystem);
354  if (_eventChannel)
355  return true;
356  else {
357  printWarn << "cannot construct Gammaawidevm event channel." << endl;
358  return false;
359  }
360  }
361 
362  if (_inputParameters->interactionType() == PHOTONPOMERONINCOHERENT) {
363  _eventChannel = new Gammaaincoherentvm(*_inputParameters, *_beamSystem);
364  if (_eventChannel)
365  return true;
366  else {
367  printWarn << "cannot construct Gammaanarrowvm event channel." << endl;
368  return false;
369  }
370  }
371 
372  printWarn << "interaction type '" << _inputParameters->interactionType() << "' "
373  << "cannot be used with particle type '" << _inputParameters->prodParticleType() << "'. "
374  << "cannot create event channel." << endl;
375  return false;
376  }
377  default:
378  {
379  printWarn << "unknown event channel '" << _inputParameters->prodParticleType() << "'."
380  << " cannot create event channel." << endl;
381  return false;
382  }
383  }
