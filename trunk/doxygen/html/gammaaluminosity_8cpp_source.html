82 void photonNucleusLuminosity::photonNucleusDifferentialLuminosity()
83 {
84  double W,dW,dY;
85  double Egamma,Y;
86  double testint,dndWdY;
87  double csgA;
88  double C;
89  int beam;
90 
91  std::string wyFileName;
92  wyFileName = _baseFileName +".txt";
93 
94  ofstream wylumfile;
95  wylumfile.precision(15);
96 
97  double bwnorm,Eth;
98 
99  dW = (_wMax-_wMin)/_nWbins;
100  dY = (_yMax-(-1.0)*_yMax)/_nYbins;
101 
102  // Write the values of W used in the calculation to slight.txt.
103  wylumfile.open(wyFileName.c_str());
104  wylumfile << getbbs().beam1().Z() <<endl;
105  wylumfile << getbbs().beam1().A() <<endl;
106  wylumfile << getbbs().beam2().Z() <<endl;
107  wylumfile << getbbs().beam2().A() <<endl;
108  wylumfile << _beamLorentzGamma <<endl;
109  wylumfile << _maxW <<endl;
110  wylumfile << _minW <<endl;
111  wylumfile << _nmbWBins <<endl;
112  wylumfile << _maxRapidity <<endl;
113  wylumfile << _nmbRapidityBins <<endl;
114  wylumfile << _productionMode <<endl;
115  wylumfile << _beamBreakupMode <<endl;
116  wylumfile << _interferenceEnabled <<endl;
117  wylumfile << _interferenceStrength <<endl;
118  wylumfile << starlightConstants::deuteronSlopePar <<endl;
119  wylumfile << _maxPtInterference <<endl;
120  wylumfile << _nmbPtBinsInterference <<endl;
121 
122  // Normalize the Breit-Wigner Distribution and write values of W to slight.txt
123  testint=0.0;
124  //Grabbing default value for C in the breit-wigner calculation
125  C=getDefaultC();
126  for(unsigned int i = 0; i <= _nWbins - 1; ++i) {
127  W = _wMin + double(i)*dW + 0.5*dW;
128  testint = testint + breitWigner(W,C)*dW;
129  wylumfile << W << endl;
130  }
131  bwnorm = 1./testint;
132 
133  // Write the values of Y used in the calculation to slight.txt.
134  for(unsigned int i = 0; i <= _nYbins - 1; ++i) {
135  Y = -1.0*_yMax + double(i)*dY + 0.5*dY;
136  wylumfile << Y << endl;
137  }
138 
139  Eth=0.5*(((_wMin+starlightConstants::protonMass)*(_wMin +starlightConstants::protonMass)-starlightConstants::protonMass*starlightConstants::protonMass)/(_protonEnergy+sqrt(_protonEnergy*_protonEnergy-starlightConstants::protonMass*starlightConstants::protonMass)));
140 
141  int A_1 = getbbs().beam1().A();
142  int A_2 = getbbs().beam2().A();
143 
144  // Do this first for the case when the first beam is the photon emitter
145  // Treat pA separately with defined beams
146  // The variable beam (=1,2) defines which nucleus is the target
147  for(unsigned int i = 0; i <= _nWbins - 1; ++i) {
148 
149  W = _wMin + double(i)*dW + 0.5*dW;
150 
151  for(unsigned int j = 0; j <= _nYbins - 1; ++j) {
152 
153  Y = -1.0*_yMax + double(j)*dY + 0.5*dY;
154 
155  if( A_2 == 1 && A_1 != 1 ){
156  // pA, first beam is the nucleus and is in this case the target
157  Egamma = 0.5*W*exp(-Y);
158  beam = 1;
159  } else if( A_1 ==1 && A_2 != 1){
160  // pA, second beam is the nucleus and is in this case the target
161  Egamma = 0.5*W*exp(Y);
162  beam = 2;
163  } else {
164  Egamma = 0.5*W*exp(Y);
165  beam = 2;
166  }
167 
168  dndWdY = 0.;
169 
170  if( Egamma > Eth && Egamma < maxPhotonEnergy() ){
171 
172  csgA=getcsgA(Egamma,W,beam);
173  dndWdY = Egamma*photonFlux(Egamma,beam)*csgA*breitWigner(W,bwnorm);
174 
175  }
176 
177  wylumfile << dndWdY << endl;
178 
179  }
180  }
181 
182  // Repeat the loop for the case when the second beam is the photon emitter.
183  // Don't repeat for pA
184  if( !( (A_2 == 1 && A_1 != 1) || (A_1 == 1 && A_2 != 1) ) ){
185 
186  for(unsigned int i = 0; i <= _nWbins - 1; ++i) {
187 
188  W = _wMin + double(i)*dW + 0.5*dW;
189 
190  for(unsigned int j = 0; j <= _nYbins - 1; ++j) {
191 
192  Y = -1.0*_yMax + double(j)*dY + 0.5*dY;
193 
194  beam = 1;
195  Egamma = 0.5*W*exp(-Y);
196 
197  dndWdY = 0.;
198 
199  if( Egamma > Eth && Egamma < maxPhotonEnergy() ){
200 
201  csgA=getcsgA(Egamma,W,beam);
202  dndWdY = Egamma*photonFlux(Egamma,beam)*csgA*breitWigner(W,bwnorm);
203 
204  }
205 
206  wylumfile << dndWdY << endl;
207 
208  }
209  }
210  }
211 
212  wylumfile << bwnorm << endl;
213  wylumfile.close();
214 
215  if(_interferenceEnabled==1)
216  pttablegen();
217 
218 }

222 void photonNucleusLuminosity::pttablegen()
223 {
224  // Calculates the pt spectra for VM production with interference
225  // Follows S. Klein and J. Nystrand, Phys. Rev Lett. 84, 2330 (2000).
226  // Written by S. Klein, 8/2002
227 
228  // fill in table pttable in one call
229  // Integrate over all y (using the same y values as in table yarray
230  // note that the cross section goes from ymin (<0) to ymax (>0), in numy points
231  // here, we go from 0 to ymax in (numy/2)+1 points
232  // numy must be even.
233 
234  // At each y, calculate the photon energies Egamma1 and Egamma2
235  // and the two photon-A cross sections
236 
237  // loop over each p_t entry in the table.
238 
239  // Then, loop over b and phi (the angle between the VM \vec{p_t} and \vec{b}
240  // and calculate the cross section at each step.
241  // Put the results in pttable
242 
243  std::string wyFileName;
244  wyFileName = _baseFileName +".txt";
245 
246  ofstream wylumfile;
247  wylumfile.precision(15);
248 
249  wylumfile.open(wyFileName.c_str(),ios::app);
250 
251  double param1pt[500],param2pt[500];
252  double *ptparam1=param1pt;
253  double *ptparam2=param2pt;
254  double dY=0.,Yp=0.,Egamma1=0.,Egamma2=0.,Wgp=0.,cs=0.,cvma=0.,Av=0.,tmin=0.,tmax=0.,ax=0.,bx=0.;
255  double csgA=0.,t=0.,sig_ga_1=0.,sig_ga_2=0.,bmax=0.,bmin=0.,db=0.,pt=0.,sum1=0.,b=0.,A1=0.,A2=0.;
256  double sumg=0.,theta=0.,amp_i_2=0.,sumint=0.;
257  int NGAUSS=0,NBIN=0;
258 
259  double xg[16]={.0483076656877383162E0,.144471961582796493E0,
260  .239287362252137075E0, .331868602282127650E0,
261  .421351276130635345E0, .506899908932229390E0,
262  .587715757240762329E0, .663044266930215201E0,
263  .732182118740289680E0, .794483795967942407E0,
264  .849367613732569970E0, .896321155766052124E0,
265  .934906075937739689E0, .964762255587506430E0,
266  .985611511545268335E0, .997263861849481564E0};
267  double ag[16]={.0965400885147278006E0, .0956387200792748594E0,
268  .0938443990808045654E0, .0911738786957638847E0,
269  .0876520930044038111E0, .0833119242269467552E0,
270  .0781938957870703065E0, .0723457941088485062E0,
271  .0658222227763618468E0, .0586840934785355471E0,
272  .0509980592623761762E0, .0428358980222266807E0,
273  .0342738629130214331E0, .0253920653092620595E0,
274  .0162743947309056706E0, .00701861000947009660E0};
275 
276  NGAUSS=16;
277 
278  //Setting input calls to variables/less calls this way.
279  double Ymax=_yMax;
280  int numy = _nYbins;
281  double Ep = _protonEnergy;
282  int ibreakup = _beamBreakupMode;
283  double NPT = _nmbPtBinsInterference;
284  double gamma_em = _beamLorentzGamma;
285  double mass = getChannelMass();
286  int beam;
287 
288  // loop over y from 0 (not -ymax) to yma
289  // changed this to go from -ymax to ymax to aid asymmetric collisions
290 
291  dY=(2.*Ymax)/numy;
292  for(int jy=1;jy<=numy;jy++){
293  Yp=-Ymax+((double(jy)-0.5)*dY);
294 
295  // Find the photon energies. Yp >= 0, so Egamma2 is smaller (no longer true if we integrate over all Y)
296  // Use the vector meson mass for W here - neglect the width
297 
298  Egamma1 = 0.5*mass*exp(Yp);
299  Egamma2 = 0.5*mass*exp(-Yp);
300 
301  // Find the sigma(gammaA) for the two directions
302  // Photonuclear Cross Section 1
303  // Gamma-proton CM energy
304  beam=2;
305 
306  Wgp=sqrt(2.*Egamma1*(Ep+sqrt(Ep*Ep-starlightConstants::protonMass*
307  starlightConstants::protonMass))+starlightConstants::protonMass*starlightConstants::protonMass);
308 
309  // Calculate V.M.+proton cross section
310 
311  cs=sqrt(16.*starlightConstants::pi*vmPhotonCoupling()*slopeParameter()*
312  starlightConstants::hbarc*starlightConstants::hbarc*sigmagp(Wgp)
313  /starlightConstants::alpha);
314  // Calculate V.M.+nucleus cross section
315  cvma=sigma_A(cs,beam);
316 
317  // Calculate Av = dsigma/dt(t=0) Note Units: fm**2/Gev**2
318 
319  Av=(starlightConstants::alpha*cvma*cvma)/(16.*starlightConstants::pi
320  *vmPhotonCoupling()*starlightConstants::hbarc*starlightConstants::hbarc);
321 
322  tmin = ((mass*mass)/(4.*Egamma1*gamma_em)*(mass*mass)/(4.*Egamma1*gamma_em));
323  tmax = tmin + 0.25;
324  ax = 0.5*(tmax-tmin);
325  bx = 0.5*(tmax+tmin);
326  csgA = 0.;
327 
328  for(int k=0;k<NGAUSS;k++){
329  t = sqrt(ax*xg[k]+bx);
330  csgA = csgA + ag[k]*getbbs().beam2().formFactor(t)*getbbs().beam2().formFactor(t);
331  t = sqrt(ax*(-xg[k])+bx);
332  csgA = csgA + ag[k]*getbbs().beam2().formFactor(t)*getbbs().beam2().formFactor(t);
333  }
334 
335  csgA = 0.5*(tmax-tmin)*csgA;
336  csgA = Av*csgA;
337  sig_ga_1 = csgA;
338 
339  // Photonuclear Cross Section 2
340  beam=1;
341 
342  Wgp=sqrt(2.*Egamma2*(Ep+sqrt(Ep*Ep-starlightConstants::protonMass*
343  starlightConstants::protonMass))+starlightConstants::protonMass*starlightConstants::protonMass);
344 
345  cs=sqrt(16.*starlightConstants::pi*vmPhotonCoupling()*slopeParameter()*
346  starlightConstants::hbarc*starlightConstants::hbarc*sigmagp(Wgp)/starlightConstants::alpha);
347 
348  cvma=sigma_A(cs,beam);
349 
350  Av=(starlightConstants::alpha*cvma*cvma)/(16.*starlightConstants::pi
351  *vmPhotonCoupling()*starlightConstants::hbarc*starlightConstants::hbarc);
352 
353  tmin = (((mass*mass)/(4.*Egamma2*gamma_em))*((mass*mass)/(4.*Egamma2*gamma_em)));
354  tmax = tmin + 0.25;
355  ax = 0.5*(tmax-tmin);
356  bx = 0.5*(tmax+tmin);
357  csgA = 0.;
358 
359  for(int k=0;k<NGAUSS;k++){
360  t = sqrt(ax*xg[k]+bx);
361  csgA = csgA + ag[k]*getbbs().beam1().formFactor(t)*getbbs().beam1().formFactor(t);
362  t = sqrt(ax*(-xg[k])+bx);
363  csgA = csgA + ag[k]*getbbs().beam1().formFactor(t)*getbbs().beam1().formFactor(t);
364  }
365 
366  csgA = 0.5*(tmax-tmin)*csgA;
367  csgA = Av*csgA;
368  sig_ga_2 = csgA;
369 
370  // Set up pttables - they find the reduction in sigma(pt)
371  // due to the nuclear form factors.
372  // Use the vector meson mass for W here - neglect width in
373  // interference calculation
374 
375  ptparam1=vmsigmapt(mass,Egamma1,ptparam1, 2);
376  ptparam2=vmsigmapt(mass,Egamma2,ptparam2, 1);
377 
378  bmin = getbbs().beam1().nuclearRadius()+getbbs().beam2().nuclearRadius();
379  // if we allow for nuclear breakup, use a slightly smaller bmin
380  if (ibreakup != 1)
381  bmin=0.95*bmin;
382 
383  // set bmax according to the smaller photon energy, following flux.f
384  if (Egamma1 >=Egamma2) {
385  bmax=bmin+6.*starlightConstants::hbarc*gamma_em/Egamma2;
386  }
387  else {
388  bmax=bmin+6.*starlightConstants::hbarc*gamma_em/Egamma1;
389  }
390  // set number of bins to a reasonable number to start
391  NBIN = 2000;
392  db = (bmax-bmin)/float(NBIN);
393  // loop over pt
394  for(int i=1;i<=NPT;i++){
395 
396  pt = (float(i)-0.5)*_ptBinWidthInterference;
397  sum1=0.0;
398  // loop over b
399  for(int j=1;j<=NBIN;j++){
400 
401  b = bmin + (float(j)-0.5)*db;
402  A1 = Egamma1*getbbs().beam1().photonDensity(Egamma1,b)*sig_ga_1*ptparam1[i];
403  A2 = Egamma2*getbbs().beam2().photonDensity(Egamma2,b)*sig_ga_2*ptparam2[i];
404  sumg=0.0;
405 
406  // do this as a Gaussian integral, from 0 to pi
407  for(int k=0;k<NGAUSS;k++){
408 
409  theta=xg[k]*starlightConstants::pi;
410  // allow for a linear sum of interfering and non-interfering amplitudes
411  amp_i_2 = A1 + A2 - 2.*_interferenceStrength
412  *sqrt(A1*A2)*cos(pt*b*cos(theta)/starlightConstants::hbarc);
413  sumg = sumg+ag[k]*amp_i_2;
414  }
415  // this is dn/dpt^2
416  // The factor of 2 is because the theta integral is only from 0 to pi
417  sumint=2.*sumg*b*db;
418  if (ibreakup > 1)
419  sumint=sumint*getbbs().probabilityOfBreakup(b);
420  sum1 = sum1 + sumint;
421 
422  }
423  // normalization is done in readDiffLum.f
424  // This is d^2sigma/dpt^2; convert to dsigma/dpt
425 
426 
427  wylumfile << sum1*pt*_ptBinWidthInterference <<endl;
428  // end of pt loop
429  }
430  // end of y loop
431  }
432  wylumfile.close();
433 }

437 double *photonNucleusLuminosity::vmsigmapt(double W, double Egamma, double *SIGMAPT, int beam)
438 {
439  //
440  // This subroutine calculates the effect of the nuclear form factor
441  // on the pt spectrum, for use in interference calculations
442  // For an interaction with mass W and photon energy Egamma,
443  // it calculates the cross section suppression SIGMAPT(PT)
444  // as a function of pt.
445  // The input pt values come from pttable.inc
446 
447 
448  double pxmax=0.,pymax=0.,dx=0.,Epom=0.,pt=0.,px0=0.,py0=0.,sum=0.,sumy=0.;
449  double py=0.,px=0.,pt1=0.,pt2=0.,f1=0.,f2=0.,q1=0.,q2=0.,norm=0.;
450  int NGAUSS =0,Nxbin=0;
451  double xg[16]={.0483076656877383162e0,.144471961582796493e0,
452  .239287362252137075e0, .331868602282127650e0,
453  .421351276130635345e0, .506899908932229390e0,
454  .587715757240762329e0, .663044266930215201e0,
455  .732182118740289680e0, .794483795967942407e0,
456  .849367613732569970e0, .896321155766052124e0,
457  .934906075937739689e0, .964762255587506430e0,
458  .985611511545268335e0, .997263861849481564e0};
459  double ag[16]={.0965400885147278006e0, .0956387200792748594e0,
460  .0938443990808045654e0, .0911738786957638847e0,
461  .0876520930044038111e0, .0833119242269467552e0,
462  .0781938957870703065e0, .0723457941088485062e0,
463  .0658222227763618468e0, .0586840934785355471e0,
464  .0509980592623761762e0, .0428358980222266807e0,
465  .0342738629130214331e0, .0253920653092620595e0,
466  .0162743947309056706e0, .00701861000947009660e0};
467  NGAUSS=16;
468 
469  // >> Initialize
470  if (beam == 1) {
471  pxmax = 10.*(starlightConstants::hbarc/getbbs().beam2().nuclearRadius());
472  pymax = 10.*(starlightConstants::hbarc/getbbs().beam2().nuclearRadius());
473  }
474  else {
475  pxmax = 10.*(starlightConstants::hbarc/getbbs().beam1().nuclearRadius());
476  pymax = 10.*(starlightConstants::hbarc/getbbs().beam1().nuclearRadius());
477  }
478 
479  Nxbin = 500;
480 
481  dx = 2.*pxmax/double(Nxbin);
482  Epom = W*W/(4.*Egamma);
483 
484  // >> Loop over total Pt to find distribution
485 
486  for(int k=1;k<=_nmbPtBinsInterference;k++){
487 
488  pt=_ptBinWidthInterference*(double(k)-0.5);
489 
490  px0 = pt;
491  py0 = 0.0;
492 
493  // For each total Pt, integrate over Pt1, , the photon pt
494  // The pt of the Pomeron is the difference
495  // pt1 is
496  sum=0.;
497  for(int i=1;i<=Nxbin;i++){
498 
499  px = -pxmax + (double(i)-0.5)*dx;
500  sumy=0.0;
501  for(int j=0;j<NGAUSS;j++){
502 
503  py = 0.5*pymax*xg[j]+0.5*pymax;
504  // photon pt
505  pt1 = sqrt( px*px + py*py );
506  // pomeron pt
507  pt2 = sqrt( (px-px0)*(px-px0) + (py-py0)*(py-py0) );
508  q1 = sqrt( ((Egamma/_beamLorentzGamma)*(Egamma/_beamLorentzGamma)) + pt1*pt1 );
509  q2 = sqrt( ((Epom/_beamLorentzGamma)*(Epom/_beamLorentzGamma)) + pt2*pt2 );
510 
511  // photon form factor
512  // add in phase space factor?
513  if (beam ==2) {
514  f1 = (getbbs().beam1().formFactor(q1*q1)*getbbs().beam1().formFactor(q1*q1)*pt1*pt1)/(q1*q1*q1*q1);
515 
516  // Pomeron form factor
517  f2 = getbbs().beam2().formFactor(q2*q2)*getbbs().beam2().formFactor(q2*q2);
518  }
519  else {
520  f1 = (getbbs().beam2().formFactor(q1*q1)*getbbs().beam2().formFactor(q1*q1)*pt1*pt1)/(q1*q1*q1*q1);
521 
522  // Pomeron form factor
523  f2 = getbbs().beam1().formFactor(q2*q2)*getbbs().beam1().formFactor(q2*q2);
524  }
525  sumy= sumy + ag[j]*f1*f2;
526 
527  // now consider other half of py phase space - why is this split?
528  py = 0.5*pymax*(-xg[j])+0.5*pymax;
529  pt1 = sqrt( px*px + py*py );
530  pt2 = sqrt( (px-px0)*(px-px0) + (py-py0)*(py-py0) );
531  q1 = sqrt( ((Egamma/_beamLorentzGamma)*Egamma/_beamLorentzGamma) + pt1*pt1 );
532  q2 = sqrt( ((Epom/_beamLorentzGamma)*(Epom/_beamLorentzGamma)) + pt2*pt2 );
533  // add in phase space factor?
534  if (beam ==2) {
535  f1 = (getbbs().beam1().formFactor(q1*q1)*getbbs().beam1().formFactor(q1*q1)*pt1*pt1)/(q1*q1*q1*q1);
536 
537  // Pomeron form factor
538  f2 = getbbs().beam2().formFactor(q2*q2)*getbbs().beam2().formFactor(q2*q2);
539  }
