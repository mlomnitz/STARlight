236 double
237 photonNucleusCrossSection::getcsgA(const double targetEgamma,
238  const double Q2,
239  const int beam)
240 {
241  //This function returns the cross-section for photon-nucleus interaction
242  //producing vectormesons
243 
244  double Av,Wgp,cs,cvma;
245  double t,tmin,tmax;
246  double csgA,ax,bx;
247  int NGAUSS;
248  //
249  double W = _channelMass; //new change, channel mass center used for the t min integration.
250 
251  // DATA FOR GAUSS INTEGRATION
252  double xg[6] = {0, 0.1488743390, 0.4333953941, 0.6794095683, 0.8650633667, 0.9739065285};
253  double ag[6] = {0, 0.2955242247, 0.2692667193, 0.2190863625, 0.1494513492, 0.0666713443};
254  NGAUSS = 6;
255 
256  // Note: The photon energy passed to this function is now in the target frame. The rest of the calculations are done in the
257  // CMS frame. The next lines boost the photon into the CM frame.
258  double E_prime = _electronEnergy - targetEgamma;
259  double cos_theta_e = 1. - Q2/(2.*_electronEnergy*E_prime);
260  double theta_e = acos(cos_theta_e);
261  double beam_y = acosh(_beamLorentzGamma);
262  double gamma_pt = E_prime*sin(theta_e);
263  double pz_squared = targetEgamma*targetEgamma - Q2 - gamma_pt*gamma_pt;
264  if( pz_squared < 0 || fabs(cos_theta_e) > 1 || 2.*targetEgamma/(Q2+W*W) < _targetRadii)
265  return 0;
266  double temp_pz = sqrt(pz_squared);
267  // Now boost to CM frame
268  double Egamma = targetEgamma*cosh(beam_y) - temp_pz*sinh(beam_y);
269  if( Egamma < _cmsMinPhotonEnergy || Egamma > _maxPhotonEnergy){
270  return 0;
271  }
272  //cout<<" ::: Lomnitz test in photonNucleus ::: pz^2 = "<<pz_squared << " CMS Egamma = "<<Egamma<<endl;
273  // Find gamma-proton CM energy in CMS frame in the limit Q2->0 (this is our assumption, the Q2 dependence is in the factor)
274  Wgp = sqrt( 2.*(protonMass*targetEgamma)+protonMass*protonMass);
275  /*Wgp = sqrt(2. * Egamma * (_protonEnergy
276  + sqrt(_protonEnergy * _protonEnergy - protonMass * protonMass))
277  + protonMass * protonMass);*/
278 
279  //Used for A-A
280  tmin = (W * W / (4. * Egamma * _beamLorentzGamma)) * (W * W / (4. * Egamma * _beamLorentzGamma));
281 
282  if ((_bbs.beam1().A() <= 1) && (_bbs.beam2().A() <= 1)){
283  // proton-proton, no scaling needed
284  csgA = getcsgA_Q2_dep(Q2)*sigmagp(Wgp);
285  } else {
286  // Check if one or both beams are nuclei
287  int A_1 = _bbs.beam1().A();
288  int A_2 = _bbs.beam2().A();
289  // coherent AA interactions
290  // Calculate V.M.+proton cross section
291  // cs = sqrt(16. * pi * _vmPhotonCoupling * _slopeParameter * hbarc * hbarc * sigmagp(Wgp) / alpha);
292  cs = getcsgA_Q2_dep(Q2)*sigma_N(Wgp); //Use member function instead
293 
294  // Calculate V.M.+nucleus cross section
295  cvma = sigma_A(cs,beam);
296 
297  // Do impulse approximation here
298  if( _impulseSelected == 1){
299  if( beam == 1 ){
300  cvma = A_1*cs;
301  } else if ( beam == 2 ){
302  cvma = A_2*cs;
303  }
304  }
305 
306  // Calculate Av = dsigma/dt(t=0) Note Units: fm**s/Gev**2
307  Av = (alpha * cvma * cvma) / (16. * pi * _vmPhotonCoupling * hbarc * hbarc);
308 
309  tmax = tmin + 0.25;
310  ax = 0.5 * (tmax - tmin);
311  bx = 0.5 * (tmax + tmin);
312  csgA = 0.;
313  for (int k = 1; k < NGAUSS; ++k) {
314 
315  t = ax * xg[k] + bx;
316  if( A_1 <= 1 && A_2 != 1){
317  csgA = csgA + ag[k] * _bbs.beam2().formFactor(t) * _bbs.beam2().formFactor(t);
318  }else if(A_2 <=1 && A_1 != 1){
319  csgA = csgA + ag[k] * _bbs.beam1().formFactor(t) * _bbs.beam1().formFactor(t);
320  }else{
321  if( beam==1 ){
322  csgA = csgA + ag[k] * _bbs.beam1().formFactor(t) * _bbs.beam1().formFactor(t);
323  }else if(beam==2){
324  csgA = csgA + ag[k] * _bbs.beam2().formFactor(t) * _bbs.beam2().formFactor(t);
325  }else{
326  cout<<"Something went wrong here, beam= "<<beam<<endl;
327  }
328  }
329 
330  t = ax * (-xg[k]) + bx;
331  if( A_1 <= 1 && A_2 != 1){
332  csgA = csgA + ag[k] * _bbs.beam2().formFactor(t) * _bbs.beam2().formFactor(t);
333  }else if(A_2 <=1 && A_1 != 1){
334  csgA = csgA + ag[k] * _bbs.beam1().formFactor(t) * _bbs.beam1().formFactor(t);
335  }else{
336  if( beam==1 ){
337  csgA = csgA + ag[k] * _bbs.beam1().formFactor(t) * _bbs.beam1().formFactor(t);
338  }else if(beam==2){
339  csgA = csgA + ag[k] * _bbs.beam2().formFactor(t) * _bbs.beam2().formFactor(t);
340  }else{
341  cout<<"Something went wrong here, beam= "<<beam<<endl;
342  }
343  }
344  }
345  csgA = 0.5 * (tmax - tmin) * csgA;
346  csgA = Av * csgA;
347  }
348  return csgA;
349 }

873 photonNucleusCrossSection::nepoint(const double Egamma,
874  const double bmin)
875 {
876  // Function for the spectrum of virtual photons,
877  // dn/dEgamma, for a point charge q=Ze sweeping
878  // past the origin with velocity gamma
879  // (=1/SQRT(1-(V/c)**2)) integrated over impact
880  // parameter from bmin to infinity
881  // See Jackson eq15.54 Classical Electrodynamics
882  // Declare Local Variables
883  double beta,X,C1,bracket,nepoint_r;
884 
885  beta = sqrt(1.-(1./(_beamLorentzGamma*_beamLorentzGamma)));
886  X = (bmin*Egamma)/(beta*_beamLorentzGamma*hbarc);
887 
888  bracket = -0.5*beta*beta*X*X*(bessel::dbesk1(X)*bessel::dbesk1(X)
889  -bessel::dbesk0(X)*bessel::dbesk0(X));
890 
891  bracket = bracket+X*bessel::dbesk0(X)*bessel::dbesk1(X);
892 
893  // Note: NO Z*Z!!
894  C1=(2.*alpha)/pi;
895 
896  nepoint_r = C1*(1./beta)*(1./beta)*(1./Egamma)*bracket;
897 
898  return nepoint_r;
