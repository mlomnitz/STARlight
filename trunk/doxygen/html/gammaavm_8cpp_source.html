221  pz2 = -pz1;

319  iFbadevent = 1;
320  return false;
321  }
322 
323  // construct parent four-vector
324  const double parentEnergy = sqrt(p[0] * p[0] + p[1] * p[1] + p[2] * p[2]
325  + parentMass * parentMass);
326  const lorentzVector parentVec(p[0], p[1], p[2], parentEnergy);
327

329  assert(_phaseSpaceGen);
330  static bool firstCall = true;
331  if (firstCall) {

333  _phaseSpaceGen->setDecay(4, m);
