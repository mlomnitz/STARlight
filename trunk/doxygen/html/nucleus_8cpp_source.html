112 double
113 nucleus::rws(const double r) const
114 {
115  if( _Z < 7 ){
116  // Gaussian density distribution for light nuclei
117  double norm = (3./(2.*starlightConstants::pi))*sqrt( (3./(2.*starlightConstants::pi)) );
118  norm = norm/(nuclearRadius()*nuclearRadius()*nuclearRadius());
119  return norm*exp(-((3./2.)*r*r)/(nuclearRadius()*nuclearRadius()));
120  }else{
121  // Fermi density distribution for heavy nuclei
122  return 1.0 / (1. + exp((r - nuclearRadius()) / woodSaxonSkinDepth()));
123  }
124 }

127 double
128 nucleus::formFactor(const double t) const
129 {
130  // electromagnetic form factor of proton
131  if ((_Z == 1) && (_A == 1)) {
132  const double rec = 1. / (1. + t / 0.71);
133  return rec * rec;
134  }
135 
136  if( _Z < 7 ){
137  // Gaussian form factor for light nuclei
138  const double R_G = nuclearRadius();
139  return exp(-R_G*R_G*t/(6.*starlightConstants::hbarc*starlightConstants::hbarc));
140  } else {
141  // nuclear form factor, from Klein Nystrand PRC 60 (1999) 014903, Eq. 14
142  const double R = nuclearRadius();
143  const double q = sqrt(t);
144  const double arg1 = q * R / hbarc;
145  const double arg2 = hbarc / (q * R);
146  const double sph = (sin(arg1) - arg1 * cos(arg1)) * 3. * arg2 * arg2 * arg2;
147  const double a0 = 0.70; // [fm]
148  return sph / (1. + (a0 * a0 * t) / (hbarc * hbarc));
149  }
150 }

162 double
163 nucleus::thickness(const double b) const
164 {
165  // JS This code calculates the nuclear thickness function as per Eq. 4 in
166  // Klein and Nystrand, PRC 60.
167  // former DOUBLE PRECISION FUNCTION T(b)
168 
169  // data for Gauss integration
170  const unsigned int nmbPoints = 5;
171  const double xg[nmbPoints + 1] = {0., 0.1488743390, 0.4333953941, 0.6794095683,
172  0.8650633667, 0.9739065285};
173  const double ag[nmbPoints + 1] = {0., 0.2955242247, 0.2692667193, 0.2190863625,
174  0.1494513492, 0.0666713443};
175 
176  const double zMin = 0;
177  const double zMax = 15;
178  const double zRange = 0.5 * (zMax - zMin);
179  const double zMean = 0.5 * (zMax + zMin);
180  double sum = 0;
181  for(unsigned int i = 1; i <= nmbPoints; ++i) {
182  double zsp = zRange * xg[i] + zMean;
183  double radius = sqrt(b * b + zsp * zsp);
184  sum += ag[i] * rws(radius);
185  zsp = zRange * (-xg[i]) + zMean;
186  radius = sqrt(b * b + zsp * zsp);
187  sum += ag[i] * rws(radius);
188  }
189 
190  return 2. * zRange * sum;
191 }
