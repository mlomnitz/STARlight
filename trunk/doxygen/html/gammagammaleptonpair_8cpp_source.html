76 void Gammagammaleptonpair::twoLeptonCrossSection()
77 {
78  //This function calculates section for 2-particle decay. For reference, see STAR Note 243, Eq. 9.
79  //calculate the 2-lepton differential cross section
80  //the 100 is to convert to barns
81  //the 2 is to account for the fact that we integrate only over one half of the rapidity range
82  //Multiply all _Farray[] by _f_max
83 
84  for(int i=0;i<_GGlepInputnumw;i++)
85  {
86  for(int j=0;j<_GGlepInputnumy;j++)
87  {
88  _sigmax[i][j]=twoMuonCrossSection(_Warray[i])*_f_max*_Farray[i][j]/(100.*_Warray[i]);
89  }
90  }
91  //calculate the total two-lepton cross section
92  double sigmasum =0.;
93  for(int i=0;i<_GGlepInputnumw-1;i++)
94  {
95  for(int j=0;j<_GGlepInputnumy-1;j++)
96  {
97  sigmasum = sigmasum +(_sigmax[i][j]+_sigmax[i+1][j]+_sigmax[i][j+1]+_sigmax[i+1][j+1])/4.*(_Yarray[j+1]-_Yarray[j])*(_Warray[i+1]-_Warray[i]);
98  }
99  }
100  //cout << "The total "<<_GGlepInputpidtest<<" cross-section is: "<<sigmasum<<" barns."<<endl;
101  cout<<endl<<endl;
102  if (sigmasum > 1.){
103  cout << "Total cross section: "<<sigmasum<<" barn."<<endl;
104  } else if (1000.*sigmasum > 1.){
105  cout << "Total cross section: "<<1000.*sigmasum<<" mb."<<endl;
106  } else if (1000000.*sigmasum > 1.){
107  cout << "Total cross section: "<<1000000.*sigmasum<<" microbarn."<<endl;
108  } else if (1.E9*sigmasum > 1.){
109  cout << "Total cross section: "<<1.E9*sigmasum<<" nanobarn."<<endl;
110  } else if (1.E12*sigmasum > 1.){
111  cout << "Total cross section: "<<1.E12*sigmasum<<" picobarn."<<endl;
112  } else {
113  cout << "Total cross section: "<<1.E15*sigmasum<<" femtobarn."<<endl;
114  }
115  cout<<endl;
116  setTotalChannelCrossSection(sigmasum);
117 
118  // Do this integration here, once per run rather than once per event (JN 220710)
119  //integrate sigma down to a function of just w
120 
121  double sgf=0.;
122 
123  for(int i=0;i<_ReadInputnumw;i++)
124  {
125  _sigofw[i]=0.;
126  for(int j=0;j<_ReadInputnumy-1;j++)
127  {
128  _sigofw[i] = _sigofw[i]+(_Yarray[j+1]-_Yarray[j])*(_sigmax[i][j+1]+_sigmax[i][j])/2.;
129  }
130  }
131 
132  //calculate the unnormalized sgfint array
133  _sigfint[0]=0.;
134  for(int i=0;i<_ReadInputnumw-1;i++)
135  {
136  sgf=(_sigofw[i+1]+_sigofw[i])*(_Warray[i+1]-_Warray[i])/2.;
137  _sigfint[i+1]=_sigfint[i]+sgf;
138  }
139 
140  //normalize sgfint array
141  _signormw=_sigfint[_ReadInputnumw-1];
142  for(int i=0;i<_ReadInputnumw;i++)
143  {
144  _sigfint[i]=_sigfint[i]/_signormw;
145  }
146  return;
147 }

151 double Gammagammaleptonpair::twoMuonCrossSection(double w)
152 {
153  //This function gives the two muon cross section as a function of Y&W.
154  //Using the formula given in G.Soff et. al Nuclear Equation of State, part B, 579
155  double s=0.,Etest=0.,deltat=0.,xL=0.,sigmuix=0.,alphasquared=0.,hbarcsquared=0.;
156  s = w*w;
157  Etest = 4.*getMass()*getMass()/s;
158  deltat = s * sqrt(1.-Etest);
159  xL = 2.*log(sqrt(s)/(2.*getMass())+sqrt(1./Etest-1.));
160  alphasquared = starlightConstants::alpha*starlightConstants::alpha;
161  hbarcsquared = starlightConstants::hbarc*starlightConstants::hbarc;
162  sigmuix = 4.*starlightConstants::pi*alphasquared/s*hbarcsquared*((1+Etest-0.5*Etest*Etest)*xL-(1./s+Etest/s)*deltat);
163  if(Etest > 1.)
164  sigmuix = 0.;
165  return sigmuix;
166 }

170 void Gammagammaleptonpair::pickw(double &w)
171 {
172 // This function picks a w for the 2-photon calculation.
173 
174  double x=0.,remainarea=0.,remainw=0.,a=0.,b=0.,c=0.;
175  int ivalw=0;
176 
177  if(_wdelta != 0)
178  {
179  w=_wdelta;
180  ivalw=_ivalwd;
181  remainw=_remainwd;
182  }
183  else{
184  //deal with the case where sigma is an array
185  //_sigofw is simga integrated over y using a linear interpolation
186  //sigint is the integral of sgfint, normalized
187 
188  //pick a random number
189  x = _randy.Rndom();
190  //compare x and sgfint to find the ivalue which is just less than the random number x
191  for(int i=0;i<_GGlepInputnumw;i++)
192  {
193  if(x > _sigfint[i]) ivalw=i;
194  }
195  //remainder above ivalw
196  remainarea = x - _sigfint[ivalw];
197 
198  //figure out what point corresponds to the excess area in remainarea
199  c = -remainarea*_signormw/(_Warray[ivalw+1]-_Warray[ivalw]);
200  b = _sigofw[ivalw];
201  a = (_sigofw[ivalw+1]-_sigofw[ivalw])/2.;
202  if(a==0.)
203  {
204  remainw = -c/b;
205  }
206  else{
207  remainw = (-b+sqrt(b*b-4.*a*c))/(2.*a);
208  }
209  _ivalwd = ivalw;
210  _remainwd = remainw;
211  //calculate the w value
212  w = _Warray[ivalw]+(_Warray[ivalw+1]-_Warray[ivalw])*remainw;
213  }
214 }

218 void Gammagammaleptonpair::picky(double &y)
219 {
220  // This function picks a y given a W
221 
222  double * sigofy;
223  double * sgfint;
224  sigofy = new double[starlightLimits::MAXYBINS];
225  sgfint = new double[starlightLimits::MAXWBINS];
226 
227  double remainw =0.,remainarea=0.,remainy=0.,a=0.,b=0.,c=0.,sgf=0.,signorm=0.,x=0.;
228  int ivalw=0,ivaly=0;
229 
230  ivalw=_ivalwd;
231  remainw=_remainwd;
232  //average over two colums to get y array
233  for(int j=0;j<_GGlepInputnumy;j++)
234  {
235  sigofy[j]=_sigmax[ivalw][j]+(_sigmax[ivalw+1][j]-_sigmax[ivalw][j])*remainw;
236  }
237 
238  //calculate the unnormalized sgfint
239  sgfint[0]=0.;
240  for(int j=0;j<_GGlepInputnumy-1;j++)
241  {
242  sgf = (sigofy[j+1]+sigofy[j])/2.;
243  sgfint[j+1]=sgfint[j]+sgf*(_Yarray[j+1]-_Yarray[j]);
244  }
245 
246  //normalize the sgfint array
247  signorm = sgfint[_GGlepInputnumy-1];
248 
249  for(int j=0;j<_GGlepInputnumy;j++)
250  {
251  sgfint[j]=sgfint[j]/signorm;
252  }
253 
254  //pick a random number
255  x = _randy.Rndom();
256  //compare x and sgfint to find the ivalue which is just less then the random number x
257  for(int i=0;i<_GGlepInputnumy;i++)
258  {
259  if(x > sgfint[i]) ivaly = i;
260  }
261  //remainder above ivaly
262  remainarea = x - sgfint[ivaly];
263 
264  //figure what point corresponds to the leftover area in remainarea
265  c = -remainarea*signorm/(_Yarray[ivaly+1]-_Yarray[ivaly]);
266  b = sigofy[ivaly];
267  a = (sigofy[ivaly+1]-sigofy[ivaly])/2.;
268  if(a==0.)
269  {
270  remainy = -c/b;
271  }
272  else{
273  remainy = (-b + sqrt(b*b-4.*a*c))/(2.*a);
274  }
275  //calculate the y value
276  y = _Yarray[ivaly]+(_Yarray[ivaly+1]-_Yarray[ivaly])*remainy;
277  delete[] sigofy;
278  delete[] sgfint;
279 }

283 void Gammagammaleptonpair::pairMomentum(double w,double y,double &E,double &px,double &py,double &pz)
284 {
285  //this function calculates px,py,pz,and E given w and y
286 
287  double anglepp1=0.,anglepp2=0.,pp1=0.,pp2=0.,E1=0.,E2=0.;
288  double pt=0.;
289 
290  //E1 and E2 are for the 2 photons in the CM frame
291  E1 = w*exp(y)/2.;
292  E2 = w*exp(-y)/2.;
293 
294  //calculate px and py
295  //to get x and y components-- phi is random between 0 and 2*pi
296  anglepp1 = _randy.Rndom();
297  anglepp2 = _randy.Rndom();
298 
299  pp1 = pp_1(E1);
300  pp2 = pp_2(E2);
301  px = pp1*cos(2.*starlightConstants::pi*anglepp1)+pp2*cos(2.*starlightConstants::pi*anglepp2);
302  py = pp1*sin(2.*starlightConstants::pi*anglepp1)+pp2*sin(2.*starlightConstants::pi*anglepp2);
303 
304  //Compute vector sum Pt=Pt1+Pt2 to find pt for the produced particle
305  pt = sqrt(px*px+py*py);
306 
307  //W is the mass of the produced particle (not necessarily on-mass-shell).Now compute its energy and pz
308  E = sqrt(w*w+pt*pt)*cosh(y);
309  pz= sqrt(w*w+pt*pt)*sinh(y);
310 
311 }

315 double Gammagammaleptonpair::pp_1(double E)
316 {
317  // This is for beam 1
318  // returns on random draw from pp(E) distribution
319  double ereds =0.,Cm=0.,Coef=0.,x=0.,pp=0.,test=0.,u=0.;
320  double singleformfactorCm=0.,singleformfactorpp1=0.,singleformfactorpp2=0.;
321  int satisfy =0;
322 
323  ereds = (E/_GGlepInputGamma_em)*(E/_GGlepInputGamma_em);
324  //sqrt(3)*E/gamma_em is p_t where the distribution is a maximum
325  Cm = sqrt(3.)*E/_GGlepInputGamma_em;
326  //the amplitude of the p_t spectrum at the maximum
327  singleformfactorCm=_bbs.beam1().formFactor(Cm*Cm+ereds);
328  Coef = 3.0*(singleformfactorCm*singleformfactorCm*Cm*Cm*Cm)/((2.*(starlightConstants::pi)*(ereds+Cm*Cm))*(2.*(starlightConstants::pi)*(ereds+Cm*Cm)));
329 
330  //pick a test value pp, and find the amplitude there
331  x = _randy.Rndom();
332  pp = x*5.*starlightConstants::hbarc/_bbs.beam1().nuclearRadius();
333  singleformfactorpp1=_bbs.beam1().formFactor(pp*pp+ereds);
334  test = (singleformfactorpp1*singleformfactorpp1)*pp*pp*pp/((2.*starlightConstants::pi*(ereds+pp*pp))*(2.*starlightConstants::pi*(ereds+pp*pp)));
335 
336  while(satisfy==0){
337  u = _randy.Rndom();
338  if(u*Coef <= test)
339  {
340  satisfy =1;
341  }
342  else{
343  x =_randy.Rndom();
344  pp = 5*starlightConstants::hbarc/_bbs.beam1().nuclearRadius()*x;
345  singleformfactorpp2=_bbs.beam1().formFactor(pp*pp+ereds);
346  test = (singleformfactorpp2*singleformfactorpp2)*pp*pp*pp/(2.*starlightConstants::pi*(ereds+pp*pp)*2.*starlightConstants::pi*(ereds+pp*pp));
347  }
348  }
349 
350  return pp;
351 }

396 void Gammagammaleptonpair::twoBodyDecay(starlightConstants::particleTypeEnum &ipid,
397  double W,
398  double px0, double py0, double pz0,
399  double& px1, double& py1, double& pz1,
400  double& px2, double& py2, double& pz2,
401  int& iFbadevent)
402 {
403  // This routine decays a particle into two particles of mass mdec,
404  // taking spin into account
405 
406  double mdec=0.,E1=0.,E2=0.;
407  double pmag, anglelep[20001];
408  double phi,theta=0.,xtest,Ecm;
409  double betax,betay,betaz;
410  double hirestheta;
411  double hiresw; //added from JN->needed precision
412 
413  mdec = getMass();
414  if(W < 2*mdec)
415  {
416  cout<<" ERROR: W="<<W<<endl;
417  iFbadevent = 1;
418  return;
419  }
420  pmag = sqrt(W*W/4. - mdec*mdec);
421 
422  // pick an orientation, based on the spin
423  // phi has a flat distribution in 2*pi
424  phi = _randy.Rndom()*2.*starlightConstants::pi;
425 
426  // find theta, the angle between one of the outgoing particles and
427  // the beamline, in the frame of the two photons
428 
429  if(getSpin()==0.5){
430  // calculate a table of integrated angle values for leptons
431  // JN05: Go from 1000->20000bins, use double precision for anglelep and thetalep. needed when W >6Gev.
432  hiresw = W;
433 
434  anglelep[0] = 0.;
435 
436  for(int i =1;i<=20000;i++)
437  {
438  hirestheta = starlightConstants::pi * double(i) /20000.;
439 
440  // Added sin(theta) phase space factor (not in thetalep) and changed E to W in thetalep call
441  // 11/9/2000 SRK
442  // Note that thetalep is form invariant, so it could be called for E, theta_lab just
443  // as well as W,theta_cm. Since there is a boost from cm to lab below, the former is fine.
444 
445  anglelep[i] = anglelep[i-1] + thetalep(hiresw,hirestheta)*sin(hirestheta);
446  }
447 
448  hirestheta = 0.;
449  xtest = _randy.Rndom();
450  for(int i =1;i<=20000;i++)
451  {
452  if(xtest > (anglelep[i]/anglelep[20000]))
453  hirestheta = starlightConstants::pi * double(i) / 20000.;
454  }
455  theta=hirestheta;
456 
457  }
458 
459  if(getSpin() != 0.5)
460  cout<<" This model cannot yet handle this spin value for lepton pairs: "<<getSpin()<<endl;
461 
462 
463  // compute unboosted momenta
464  px1 = sin(theta)*cos(phi)*pmag;
465  py1 = sin(theta)*sin(phi)*pmag;
466  pz1 = cos(theta)*pmag;
467  px2 = -px1;
468  py2 = -py1;
469  pz2 = -pz1;
470 
471  // compute energies
472  //Changed mass to W 11/9/2000 SRK
473  Ecm = sqrt(W*W+px0*px0+py0*py0+pz0*pz0);
474 
475  E1 = sqrt(mdec*mdec+px1*px1+py1*py1+pz1*pz1);
476  E2 = sqrt(mdec*mdec+px2*px2+py2*py2+pz2*pz2);
477  // decay tau to electrons
478  // note that after this routine px1, etc., refer to the electrons
479  if(_GGlepInputpidtest == starlightConstants::TAUONDECAY)
480  tauDecay(px1,py1,pz1,E1,px2,py2,pz2,E2);
481 
482  // Lorentz transform into the lab frame
483  // betax,betay,betaz are the boost of the complete system
484  betax = -(px0/Ecm);
485  betay = -(py0/Ecm);
486  betaz = -(pz0/Ecm);
487 
488  transform (betax,betay,betaz,E1,px1,py1,pz1,iFbadevent);
489  transform (betax,betay,betaz,E2,px2,py2,pz2,iFbadevent);
490 
491 
492  if(iFbadevent == 1)
493  return;
494 
495  // change particle id from that of parent to that of daughters
496  // change taoun id into electron id, already switched particles in tau decay
497  if(_GGlepInputpidtest == starlightConstants::TAUONDECAY)
498  ipid = starlightConstants::ELECTRON;
499  // electrons remain electrons; muons remain muons
500  if ( (_GGlepInputpidtest == starlightConstants::ELECTRON) || (_GGlepInputpidtest == starlightConstants::MUON) ||
501  (_GGlepInputpidtest == starlightConstants::TAUON) )
502  ipid = _GGlepInputpidtest;
503 }

506 //______________________________________________________________________________
507 double Gammagammaleptonpair::thetalep(double W,double theta)
508 {
509  // This function calculates the cross-section as a function of
510  // angle for a given W and Y, for the production of two muons.
511  // (or tauons)
512  // expression is taken from Brodsky et al. PRD 1971, 1532
513  // equation 5.7
514  // factor that are not dependant on theta are scrapped, so the
515  // the absolute crosssections given by this function are inaccurate
516  // here we are working in the CM frame of the photons and the last
517  // term is 0
518 
519  // real function thetalep (W,theta)
520 
521  double moverw=0., W1sq=0.;
522  double thetalep_r=0.,denominator=0.;
523 
524  W1sq = (W / 2.)*(W/2.);
525  moverw = getMass()*getMass() / W1sq;
526  denominator = (1. - (1. - moverw)*(cos(theta)*cos(theta)));
527 
528  thetalep_r = 2. + 4.*(1.-moverw)*((1.-moverw)*sin(theta)*sin(theta)*cos(theta)*cos(theta) + moverw) / (denominator*denominator);
529 
530  return thetalep_r;
531 
532 }

547  //this function decays particles and writes events to a file
548  //zero out the event structure

565  if (iFbadevent==0){

688 void Gammagammaleptonpair::calculateTable()
689 {
690  // this subroutine calculates the tables that are used
691  // elsewhere in the montecarlo
692  // the tauon decay is taken from V-A theory, 1 - 1/3 cos(theta)
693  // the energy of each of the two leptons in tau decay
694  // is calculated using formula 10.35 given
695  // in introduction to elementary particles by D. griffiths
696  // which assmes that the mass of the electron is 0.
697  // the highest energy attainable by the electron in such a system is
698  // .5 * mass of the tau
699 
700  // subroutine calculateTable
701 
702 
703  double E,theta;
704 
705  _tautolangle[0] = 0.;
706  _dgammade[0] = 0.;
707 
708 
709  for(int i =1;i<=100;i++)
710  {
711  // calculate energy of tau decay
712  E = double(i)/100. * .5 * starlightConstants::tauMass;
713  _dgammade[i] = _dgammade[i-1] + E*E * (1. - 4.*E/(3.*starlightConstants::tauMass));
714 
715  // calculate angles for tau
716  theta = starlightConstants::pi * double(i) / 100.;
717  _tautolangle[i] = _tautolangle[i-1] + (1 + 0.3333333 * cos(theta));
718  }
719

721 }

725 void Gammagammaleptonpair::tauDecay(double &px1,double &py1,double &pz1,double &E1,double &px2,double &py2,double &pz2,double &E2)
726 {
727  // this routine assumes that the tauons decay to electrons and
728  // calculates the directons of the decays
729 
730  double Ee1,Ee2,theta1,theta2,phi1,phi2, ran1, ran2 ;
731  double pmag1,pex1,pey1,pez1,pex2,pey2,pez2,pmag2;
732  double betax,betay,betaz,dir;
733 
734  int Tmp_Par=0; // temp variable for the transform function .. kind of starnge - being called with 7 parameter instead of 8
735 
736  // the highest energy attainable by the electron in this system is
737  // .5 * mass of the tau
738 
739  // get two random numbers to compare with
740 
741 
742  ran1 = _randy.Rndom()*_dgammade[100];
743  ran2 = _randy.Rndom()*_dgammade[100];
744 
745  // compute the energies that correspond to those numbers
746  Ee1 = 0.;
747  Ee2 = 0.;
748 
749  for( int i =1;i<=100;i++)
750  {
751  if (ran1 > _dgammade[i])
752  Ee1 = double(i) /100. * .5 * getMass();
753  if (ran2 > _dgammade[i])
754  Ee2 = double(i) /100. * .5 * getMass();
755  }
756 
757  // to find the direction of emmission, first
758  // we determine if the tauons have spin of +1 or -1 along the
759  // direction of the beam line
760  dir = 1.;
761  if ( _randy.Rndom() < 0.5 )
762  dir = -1.;
763 
764  // get two random numbers to compare with
765  ran1 = _randy.Rndom()*_tautolangle[100];
766  ran2 = _randy.Rndom()*_tautolangle[100];
767 
768  // find the angles corrsponding to those numbers
769  theta1 = 0.;
770  theta2 = 0.;
771  for( int i =1;i<=100;i++)
772  {
773  if (ran1 > _tautolangle[i]) theta1 = starlightConstants::pi * double(i) /100.;
774  if (ran2 > _tautolangle[i]) theta2 = starlightConstants::pi * double(i) /100.;
775  }
776 
777  // grab another two random numbers to determine phi's
778  phi1 = _randy.Rndom()*2.*starlightConstants::pi;
779  phi2 = _randy.Rndom()*2.*starlightConstants::pi;
780  // figure out the momenta of the electron in the frames of the
781  // tauons from which they decayed, that is electron1 is in the
782  // rest frame of tauon1 and e2 is in the rest fram of tau2
783  // again the electrons are assumed to be massless
784  pmag1 = Ee1;
785  pex1 = cos(phi1)*sin(theta1)*pmag1;
786  pey1 = sin(phi1)*sin(theta1)*pmag1;
787  pez1 = cos(theta1)*pmag1*dir;
788  pmag2 = Ee2;
789  pex2 = cos(phi2)*sin(theta2)*pmag2;
790  pey2 = sin(phi2)*sin(theta2)*pmag2;
791  pez2 = cos(theta2)*pmag2*(-1.*dir);
792  // now Lorentz transform into the frame of each of the particles
793  // do particle one first
794  betax = -(px1/E1);
795  betay = -(py1/E1);
796  betaz = -(pz1/E1);
797  //cout<<"2decay betax,pex1= "<<betax<<" "<<pex1<<endl;
798  transform (betax,betay,betaz,Ee1,pex1,pey1,pez1,Tmp_Par);
799  // then do particle two
800  betax = -(px1/E2);
801  betay = -(py1/E2);
802  betaz = -(pz1/E2);
803 
804  transform (betax,betay,betaz,Ee2,pex2,pey2,pez2, Tmp_Par);
805  // finally dump the electron values into the approriate
806  // variables
807  E1 = Ee1;
808  E2 = Ee2;
809  px1 = pex1;
810  px2 = pex2;
811  py1 = pey1;
812  py2 = pey2;
813  pz1 = pez1;
814  pz2 = pez2;
815 }
