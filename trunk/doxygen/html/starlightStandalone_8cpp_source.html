110 bool
111 starlightStandalone::run()
112 {
113  if (!_starlight) {
114  printWarn << "null pointer to starlight object. make sure that init() was called. "
115  << "cannot generate events." << endl;
116  return false;
117  }
118 
119  // open output file
120  eventFileWriter fileWriter;
121  fileWriter.writeFullPythiaInfo(_inputParameters->pythiaFullEventRecord());
122  _baseFileName = _inputParameters->baseFileName();
123  _eventDataFileName = _baseFileName +".out";
124  fileWriter.open(_eventDataFileName);
125 
126  printInfo << "generating events:" << endl;
127  unsigned int nmbEvents = 0;
128  while (nmbEvents < _nmbEventsTot) {
129  for (unsigned int iEvent = 0; (iEvent < _nmbEventsPerFile) && (nmbEvents < _nmbEventsTot);
130  ++iEvent, ++nmbEvents) {
131  progressIndicator(iEvent, _nmbEventsTot, true, 4);
132  upcEvent event = _starlight->produceEvent();
133  // Boost event from CMS system to lab system
134  boostEvent(event);
135  fileWriter.writeEvent(event, iEvent);
136  }
137  }
138  fileWriter.close();
139 
140  if( _starlight->nmbAttempts() == 0 )return true;
141 
142  double _branchingRatio = _inputParameters->inputBranchingRatio();
143  printInfo << "number of attempts = " << _starlight->nmbAttempts() << ", "
144  << "number of accepted events = " << _starlight->nmbAccepted() << endl;
145  double selectedCrossSection =
146  ((double)_starlight->nmbAccepted()/_starlight->nmbAttempts())*_branchingRatio*_starlight->getTotalCrossSection();
147  if (selectedCrossSection > 1.){
148  cout<< " The cross section of the generated sample is "<<selectedCrossSection<<" barn."<<endl;
149  } else if (1.E3*selectedCrossSection > 1.){
150  cout<< " The cross section of the generated sample is "<<1.E3*selectedCrossSection<<" mb."<<endl;
151  } else if (1.E6*selectedCrossSection > 1.){
152  cout<< " The cross section of the generated sample is "<<1.E6*selectedCrossSection<<" microbarn."<<endl;
153  } else if (1.E9*selectedCrossSection > 1.){
154  cout<< " The cross section of the generated sample is "<<1.E9*selectedCrossSection<<" nanobarn."<<endl;
155  } else if (1.E12*selectedCrossSection > 1.){
156  cout<< " The cross section of the generated sample is "<<1.E12*selectedCrossSection<<" picobarn."<<endl;
157  } else {
158  cout<< " The cross section of the generated sample is " <<1.E15*selectedCrossSection<<" femtob."<<endl;
159  }
160 
161  return true;
162 }
163 void starlightStandalone::boostEvent(upcEvent &event)
164 {
165 
166  // Calculate CMS boost
167  double rap1 = acosh(_inputParameters->beam1LorentzGamma());
168  double rap2 = -acosh(_inputParameters->beam2LorentzGamma());
169  double boost = (rap1+rap2)/2.;
170 
171  event.boost(boost);
172 }
